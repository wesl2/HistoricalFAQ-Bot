### 前端技术选型

| 方案 | 优点 | 缺点 | 本项目选择 |
|------|------|------|-----------|
| **React/Vue** | 组件化、生态好 | 需构建、复杂度高 | ❌ |

1. **零配置**: CDN 引入即可使用

3. **响应式**: 内置断点系统
4. **主题定制**: 轻松修改配色

### 1. 核心架构升级：标准 RAG 系统与 PGVector 检索器

**❌ 旧版痛点**：使用 `LangChainIntegration` 单一大类管理所有逻辑，向量库强依赖 `Chroma/FAISS`，导致数据割裂。且检索器在 LCEL 中传参易错。

**✅ 新版标准（当前代码）**：拆分为独立的 `PGVectorRetriever`（继承 LangChain `BaseRetriever`）和 `StandardRAGSystem`（统一入口）。所有知识库数据全部存入 PostgreSQL。

> **重要变更**：`src/rag/langchain_integration.py` 已被废弃（文件已备份/删除），取而代之的是标准模块：`standard_rag.py`、`standard_chain.py`、`standard_retriever.py`、`standard_memory.py`、`standard_streaming.py`。

```python
# src/rag/standard_rag.py (统一入口)

from src.llm.standard_llm import get_standard_llm
from src.rag.standard_retriever import get_pgvector_retriever
from src.rag.standard_chain import build_standard_rag_chain, build_conversational_rag_chain
from src.rag.standard_memory import get_standard_memory

class StandardRAGSystem:
    """标准 RAG 系统（统一入口）
    
    公司实践：
    - 使用 LangChain 标准接口
    - LCEL 组合式写法
    - 清晰的流式输出
    - 外部 Memory 管理
    """
    
    def __init__(
        self,
        llm_mode: str = None,
        session_id: str = None,
        use_postgres_memory: bool = True,
        retriever_search_type: str = "hybrid",  # "vector", "faq", "hybrid"
    ):
        # 1. 标准 LLM (BaseChatModel)
        self.llm = get_standard_llm(llm_mode)
        
        # 2. PGVector 检索器 (不再使用 Chroma/FAISS)
        self.retriever = get_pgvector_retriever(search_type=retriever_search_type)
        
        # 3. PostgreSQL 持久化对话历史
        self.memory = None
        if session_id:
            self.memory = get_standard_memory(session_id, use_postgres_memory)

    def query(self, question: str) -> str:
        """标准 RAG 查询"""
        chain = build_standard_rag_chain(llm=self.llm, retriever=self.retriever)
        result = chain.invoke(question)
        
        if self.memory:
            self.memory.add_user_message(question)
            self.memory.add_ai_message(result)
        return result
    
    def query_with_history(self, question: str) -> str:
        """带对话历史的查询（标准 Conversational RAG）
        
        使用 MessagesPlaceholder 传递结构化历史消息，
        而不是拼接字符串，确保 LCEL 链正确工作。
        """
        if not self.memory:
            return self.query(question)
        
        chain = build_conversational_rag_chain(
            llm=self.llm,
            retriever=self.retriever,
            use_history_rewrite=True,
        )
        
        # 修复：传递标准的 dict，包含 history messages list 和 question
        history_messages = self.memory.get_messages()
        result = chain.invoke({
            "history": history_messages,
            "question": question,
        })
        
        self.memory.add_user_message(question)
        self.memory.add_ai_message(result)
        return result
```

### 2. PGVector 检索器 (`src/rag/standard_retriever.py`)

这是重构的核心之一，我们**不再使用 Chroma/FAISS**，而是直接复用项目已有的 `pg_search` 模块并封装成 LangChain 标准接口，支持 **RRF 融合排序**：

```python
# src/rag/standard_retriever.py

from langchain_core.retrievers import BaseRetriever
from langchain_core.documents import Document
from langchain_huggingface import HuggingFaceEmbeddings
from src.vectorstore.pg_search import FAQSearcher, HybridSearcher
from config.model_config import EMBEDDING_CONFIG
import hashlib
import re
from collections import defaultdict


def _hash_text(t: str) -> str:
    """计算文本 SHA1 指纹，用于 RRF 融合时的去重与对齐"""
    t = (t or "").strip().lower()
    t = re.sub(r"\s+", " ", t)
    return hashlib.sha1(t.encode("utf-8")).hexdigest()


class PGVectorRetriever(BaseRetriever):
    """PGVector 检索器，完全基于 PostgreSQL + pgvector"""
    
    search_type: str = "hybrid"  # "faq", "doc", "hybrid"
    search_kwargs: dict = {"top_k": 5}
    _embeddings: Any = None  # 缓存 embedding 实例

    def _get_relevant_documents(self, query: str) -> list[Document]:
        embeddings = self._get_embeddings()
        query_vector = embeddings.embed_query(query)

        if self.search_type == "faq":
            return self._search_faq(query_vector)
        elif self.search_type == "hybrid":
            return self._search_hybrid(query_vector, query)
        else:
            return self._search_docs(query_vector)
    
    def _search_hybrid(self, query_vector: List[float], query_text: str) -> List[Document]:
        """
        混合检索（双路召回 + RRF 融合排序 + 跨源去重）
        
        工业界实践：
        1. 扩大召回：每路召回 Top-K * 2
        2. RRF 融合：基于内容 Hash 对齐 FAQ 和 Doc，计算 1/(K+rank)
        3. 智能去重：优先保留结构化更好的 FAQ
        """
        top_k = self.search_kwargs.get("top_k", 5)
        recall_k = top_k * 2  # 扩大召回因子

        hybrid_searcher = HybridSearcher()
        results = hybrid_searcher.hybrid_search(query_vector, query_text, top_k=recall_k)
        
        faq_items = results.get("faq", [])
        doc_items = results.get("doc", [])

        # ========== RRF 融合计算 ==========
        K = 60  # 经典推荐值
        rrf_scores = defaultdict(float)
        best_item = {}  # key -> 代表 Item (优先存 FAQ)
        ranks = {}      # key -> {faq_rank, doc_rank}

        # FAQ 排名贡献
        for idx, r in enumerate(faq_items, start=1):
            key = _hash_text(r.answer)
            rrf_scores[key] += 1.0 / (K + idx)
            ranks.setdefault(key, {})["faq_rank"] = idx
            
            if key not in best_item:  # FAQ 优先
                best_item[key] = ("faq", r)

        # Doc 排名贡献
        for idx, r in enumerate(doc_items, start=1):
            key = _hash_text(r.answer)
            rrf_scores[key] += 1.0 / (K + idx)
            ranks.setdefault(key, {})["doc_rank"] = idx
            
            if key not in best_item:  # FAQ 没出现过才存
                best_item[key] = ("doc", r)

        # 统一排序 & 截断
        sorted_keys = sorted(rrf_scores.keys(), key=lambda k: rrf_scores[k], reverse=True)
        
        docs = []
        for k in sorted_keys[:top_k]:
            source_type, r = best_item[k]
            
            meta = {
                "type": source_type,
                "id": r.id,
                "source": getattr(r, "source_doc", None) or "unknown",
                "rrf_score": rrf_scores[k],
                "faq_rank": ranks.get(k, {}).get("faq_rank"),
                "doc_rank": ranks.get(k, {}).get("doc_rank"),
                "content_hash": k,
            }
            
            if source_type == "faq":
                meta["question"] = getattr(r, "question", "")
            if source_type == "doc":
                meta["page"] = getattr(r, "source_page", None)
            
            docs.append(Document(page_content=r.answer, metadata=meta))

        return docs
```

```python
# src/rag/standard_chain.py

from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.output_parsers import StrOutputParser
from langchain_core.runnables import RunnablePassthrough, RunnableParallel
from operator import itemgetter

def format_docs(docs: List[Document]) -> str:
    """格式化检索结果为上下文字符串"""
    if not docs:
        return "无相关参考资料"
    return "\n\n".join(
        f"来源：{doc.metadata.get('source', '未知')}\n内容：{doc.page_content}"
        for doc in docs
    )

def build_standard_rag_chain(llm=None, retriever=None):
    """标准 RAG Chain（LCEL 写法）"""
    llm = llm or get_standard_llm()
    retriever = retriever or get_pgvector_retriever()
    
    prompt = ChatPromptTemplate.from_messages([
        ("system", """你是一位专业的历史研究专家。
请基于提供的参考资料回答用户问题。

参考资料：
{context}"""),
        ("human", "{question}")
    ])
    
    # LCEL Chain
    rag_chain = (
        RunnableParallel({
            "context": retriever | format_docs,
            "question": RunnablePassthrough()
        })
        | prompt
        | llm
        | StrOutputParser()
    )
    
    return rag_chain


def build_conversational_rag_chain(llm=None, retriever=None, use_history_rewrite=True):
    """带对话历史的 RAG Chain"""
    llm = llm or get_standard_llm()
    retriever = retriever or get_pgvector_retriever()
    
    if use_history_rewrite:
        # 1. 问题重写 Prompt
        rewrite_prompt = ChatPromptTemplate.from_messages([
            ("system", "你是一个问题重写助手..."),
            MessagesPlaceholder(variable_name="history"),
            ("human", "{question}")
        ])
        
        # 2. RAG Prompt
        rag_prompt = ChatPromptTemplate.from_messages([
            ("system", "你是一位专业的历史研究专家...\n\n参考资料：\n{context}"),
            ("human", "{rewritten_question}")
        ])
        
        # 3. 构建 Chain
        rewrite_chain = rewrite_prompt | llm | StrOutputParser()
        
        # 关键：使用 itemgetter 确保 retriever 收到 rewritten_question 字符串
        rag_chain = (
            RunnableParallel({
                "rewritten_question": rewrite_chain,
                "context": itemgetter("rewritten_question") | retriever | format_docs,
                "history": RunnablePassthrough()
            })
            | rag_prompt
            | llm
            | StrOutputParser()
        )
        
        return rag_chain
```

### Multi-Query 检索原理

```
用户查询："王洪文的生平"
    │
    ▼
LLM 生成变体
    ├── "王洪文是谁，他的生平如何"
    ├── "介绍一下王洪文的历史"
    └── "王洪文的个人经历"
    │
    ▼
并行检索 (3 个查询同时检索)
    │
    ▼
合并去重 → Top-K 结果
```

**优势**: 克服单一查询的语义局限，提升召回率

### Rerank 重排序原理

```
第一阶段（召回）          第二阶段（精排）
向量相似度检索           交叉编码器评分
    ↓                        ↓
Top 20 文档              Top 5 文档
    │                        │
    └──────→ 重排序 ────────→┘
```

**为什么需要 Rerank？**
- 向量检索快但精度有限
- 交叉编码器慢但精度高
- 两阶段结合：效率 + 精度

### BCE-Reranker 模型

| 特性 | 说明 |
|------|------|
| **模型名** | BCE-Reranker-Base_v1 |
| **类型** | Cross-Encoder |
| **输入** | (Query, Passage) 对 |
| **输出** | 相关性分数 (0-1) |
| **优势** | 中文优化、精度高 |

```python
# -*- coding: utf-8 -*-
"""高级检索模块"""

import logging
from typing import List
from pathlib import Path

from langchain_core.documents import Document
from langchain.retrievers.multi_query import MultiQueryRetriever
from langchain_community.vectorstores import Chroma

from src.llm.llm_factory import LLMFactory
from config.model_config import EMBEDDING_CONFIG, RERANKER_CONFIG, LANGCHAIN_CONFIG

logger = logging.getLogger(__name__)


class BCEReranker:
    """BCE-Reranker 重排序器"""
    
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._initialized = False
        return cls._instance
    
    def __init__(self):
        if self._initialized:
            return
        
        try:
            from BCEmbedding import RerankerModel
            
            self.model = RerankerModel(
                model_name_or_path=RERANKER_CONFIG["model_path"],
                device=RERANKER_CONFIG["device"],
                use_fp16=RERANKER_CONFIG["use_fp16"]
            )
            self.top_n = RERANKER_CONFIG["top_n"]
            self._initialized = True
            
            logger.info(f"BCE-Reranker 初始化完成")
        except Exception as e:
            logger.error(f"BCE-Reranker 初始化失败：{e}")
            self.model = None
    
    def rerank(self, query: str, documents: List[Document], 
               top_n: int = None) -> List[Document]:
        """
        重排序
        
        Args:
            query: 查询文本
            documents: 待排序文档
            top_n: 返回数量
        
        Returns:
            重排序后的文档
        """
        if self.model is None or not documents:
            return documents
        
        top_n = top_n or self.top_n
        
        try:
            # 准备数据
            passages = [doc.page_content for doc in documents]
            
            # 计算相关性分数
            scores = self.model.compute_score(query, passages)
            
            # 按分数排序
            doc_score_pairs = list(zip(documents, scores))
            doc_score_pairs.sort(key=lambda x: x[1], reverse=True)
            
            # 返回 top_n
            return [doc for doc, score in doc_score_pairs[:top_n]]
        
        except Exception as e:
            logger.error(f"重排序失败：{e}")
            return documents[:top_n]


class AdvancedRetriever:
    """高级检索器"""
    
    def __init__(self, vectorstore_path: str = None,
                 use_multi_query: bool = True,
                 use_rerank: bool = True,
                 llm_mode: str = None):
        """
        初始化
        
        Args:
            vectorstore_path: 向量库持久化路径
            use_multi_query: 是否使用多查询
            use_rerank: 是否使用重排序
            llm_mode: LLM 模式
        """
        self.vectorstore_path = vectorstore_path or LANGCHAIN_CONFIG["vectorstore"]["persist_directory"]
        self.use_multi_query = use_multi_query
        self.use_rerank = use_rerank
        
        # 初始化 LLM
        self.llm = LLMFactory.create_llm(llm_mode)
        
        # 初始化 Reranker
        self.reranker = BCEReranker() if use_rerank else None
        
        # 向量库和检索器
        self.vectorstore = None
        self.retriever = None
        
        # 加载持久化向量库
        self._load_vectorstore()
        
        logger.info(f"高级检索器初始化完成：multi_query={use_multi_query}, rerank={use_rerank}")
    
    def _load_vectorstore(self):
        """加载持久化的向量库"""
        try:
            from langchain_community.embeddings import HuggingFaceEmbeddings
            
            embeddings = HuggingFaceEmbeddings(
                model_name=EMBEDDING_CONFIG["model_path"],
                model_kwargs={"device": EMBEDDING_CONFIG["device"]},
                encode_kwargs={"normalize_embeddings": EMBEDDING_CONFIG["normalize"]}
            )
            
            # 检查是否存在持久化的向量库
            chroma_path = Path(self.vectorstore_path) / "chroma"
            if chroma_path.exists():
                self.vectorstore = Chroma(
                    persist_directory=str(chroma_path),
                    embedding_function=embeddings
                )
                self._init_retriever()
                logger.info(f"加载持久化向量库成功")
        except Exception as e:
            logger.warning(f"加载持久化向量库失败：{e}")
    
    def _init_retriever(self):
        """初始化检索器"""
        if self.vectorstore is None:
            return
        
        # 基础检索器
        base_retriever = self.vectorstore.as_retriever(
            search_type="similarity",
            search_kwargs={"k": 10}  # 召回更多，给 rerank 留空间
        )
        
        # 包装为 Multi-Query 检索器
        if self.use_multi_query:
            self.retriever = MultiQueryRetriever.from_llm(
                retriever=base_retriever,
                llm=self.llm,
                prompt=self._load_multi_query_prompt()
            )
            logger.info("Multi-Query 检索器初始化完成")
        else:
            self.retriever = base_retriever
    
    def _load_multi_query_prompt(self):
        """加载 multi-query 提示词"""
        from langchain_core.prompts import PromptTemplate
        
        prompt_path = Path(__file__).parent.parent.parent / "prompts" / "multi_query_template.txt"
        
        if prompt_path.exists():
            with open(prompt_path, "r", encoding="utf-8") as f:
                template = f.read()
        else:
            template = """你是一个 AI 语言模型助手。你的任务是生成给定用户问题的 3 个不同版本。
请提供以换行分隔的替代问题。

原始问题：{question}

替代问题:"""
        
        return PromptTemplate(template=template, input_variables=["question"])
    
    def retrieve(self, query: str, k: int = 5, 
                 rerank_top_n: int = None) -> List[Document]:
        """
        检索文档
        
        Args:
            query: 查询文本
            k: 检索数量
            rerank_top_n: 重排序后返回数量
        
        Returns:
            检索到的文档
        """
        if self.retriever is None:
            logger.error("检索器未初始化")
            return []
        
        try:
            # 第一阶段：检索（multi-query 会生成多个查询）
            docs = self.retriever.invoke(query)
            logger.info(f"第一阶段检索完成：召回 {len(docs)} 个文档")
            
            # 第二阶段：重排序
            if self.use_rerank and self.reranker:
                docs = self.reranker.rerank(query, docs, top_n=(rerank_top_n or k))
            else:
                docs = docs[:k]
            
            return docs
        
        except Exception as e:
            logger.error(f"检索失败：{e}")
            return []
    
    def create_vectorstore(self, documents: List[Document], 
                          vectorstore_type: str = "chroma"):
        """
        创建向量库并持久化
        
        Args:
            documents: 文档列表
            vectorstore_type: 向量库类型
        """
        from langchain_community.embeddings import HuggingFaceEmbeddings
        
        embeddings = HuggingFaceEmbeddings(
            model_name=EMBEDDING_CONFIG["model_path"],
            model_kwargs={"device": EMBEDDING_CONFIG["device"]},
            encode_kwargs={"normalize_embeddings": EMBEDDING_CONFIG["normalize"]}
        )
        
        if vectorstore_type == "chroma":
            persist_dir = Path(self.vectorstore_path) / "chroma"
            persist_dir.mkdir(parents=True, exist_ok=True)
            
            self.vectorstore = Chroma.from_documents(
                documents=documents,
                embedding=embeddings,
                persist_directory=str(persist_dir)
            )
            self.vectorstore.persist()
            
            self._init_retriever()
            logger.info(f"Chroma 向量库创建并持久化：{persist_dir}")


def get_advanced_retriever(vectorstore_path: str = None,
                           use_multi_query: bool = True,
                           use_rerank: bool = True):
    """工厂函数：获取高级检索器"""
    return AdvancedRetriever(
        vectorstore_path=vectorstore_path,
        use_multi_query=use_multi_query,
        use_rerank=use_rerank
    )
```

### SSE 工作原理

```
客户端                          服务端
  │                              │
  │  ── HTTP GET /stream ──→    │
  │                              │
  │  ← data: 你 ──────────────   │
  │  ← data: 好 ──────────────   │
  │  ← data: ， ──────────────   │
  │  ← data: 我 ──────────────   │
  │  ← data: 是 ──────────────   │
  │  ← data: [DONE] ─────────    │
  │                              │
  └────── 连接保持 ──────────────┘
```

```
data: 第一行内容\n\n
```

```python
from typing import Generator, AsyncGenerator

class ChatEngine:
    # ... 其他方法
    
    def stream(self, query: str) -> Generator[str, None, None]:
        """
        流式生成回答（同步）
        
        Args:
            query: 查询文本
        
        Yields:
            生成的文本片段
        """
        if not self.use_langchain or not self.langchain:
            raise ValueError("流式输出需要 LangChain 支持")
        
        # 检索
        search_context = self.search_router.search(query)
        
        if search_context.search_type == SearchType.FAQ_ONLY:
            # FAQ 直接返回，模拟流式
            answer = search_context.faq_results[0].answer
            import re
            sentences = re.split(r'([.!?;\n])', answer)
            buffer = ""
            for s in sentences:
                buffer += s
                if s in ['。', '！', '？', '；', '\n']:
                    yield buffer
                    buffer = ""
            if buffer:
                yield buffer
        else:
            # LangChain 流式生成
            documents = self._build_documents(
                search_context.faq_results,
                search_context.doc_results
            )
            retriever = self._get_or_create_retriever(documents)
            
            for chunk in self.langchain.stream(query):
                yield chunk
    
    async def astream(self, query: str) -> AsyncGenerator[str, None]:
        """
        异步流式生成
        
        Args:
            query: 查询文本
        
        Yields:
            生成的文本片段
        """
        if not self.use_langchain or not self.langchain:
            raise ValueError("流式输出需要 LangChain 支持")
        
        async for chunk in self.langchain.astream(query):
            yield chunk
```

```javascript
async function streamQueryWithFetch(question) {
    const response = await fetch(`${API_URL}/api/query/stream`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ question })
    });
    
    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    let buffer = '';
    
    while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        
        buffer += decoder.decode(value, { stream: true });
        
        // 解析 SSE 格式
        const lines = buffer.split('\n');
        buffer = lines.pop() || '';  // 保留不完整行

            if (line.startsWith('data: ')) {
                const data = line.slice(6);
                if (data === '[DONE]') {
                    return;
                }
                // 处理数据

        }
    }
```

### 可观测性三支柱

```
┌─────────────────────────────────────────┐
│          可观测性系统                    │
├─────────────────────────────────────────┤
│                                          │
│  1. Logs (日志)     - 发生了什么？      │
│  2. Metrics (指标)  - 系统状态？        │
│  3. Traces (追踪)   - 执行流程？        │
│                                          │
└─────────────────────────────────────────┘
```

### LangChain Callback 机制

```
Chain 执行流程
    │
    ├── on_chain_start    → 记录开始
    ├── on_llm_start      → LLM 调用
    ├── on_llm_end        → LLM 返回
    ├── on_retriever_end  → 检索完成
    └── on_chain_end      → 记录结束
```

### 监控指标设计

| 指标类型 | 具体指标 |
|----------|----------|
| **性能** | Chain 数量、LLM 调用次数、平均延迟 |
| **Token** | 总 Token 数、输入/输出 Token、成本估算 |
| **质量** | 检索文档数、重排序分数 |

### 1. Callback 处理器 (src/rag/callbacks.py)

```python
# -*- coding: utf-8 -*-
"""Callback 可观测性系统"""

import json
import time
import logging
from typing import Any, Dict, List
from datetime import datetime
from pathlib import Path

from langchain_core.callbacks import BaseCallbackHandler
from langchain_core.outputs import LLMResult
from langchain_core.documents import Document

logger = logging.getLogger(__name__)


class LoggingCallbackHandler(BaseCallbackHandler):
    """日志回调处理器"""
    
    def __init__(self, log_file: str = None):
        super().__init__()
        self.log_file = log_file or f"logs/callback_{datetime.now().strftime('%Y%m%d')}.jsonl"
        self.logs = []
        self.start_times = {}
        
        # 确保日志目录存在
        Path(self.log_file).parent.mkdir(parents=True, exist_ok=True)
    
    def _log(self, event_type: str, data: Dict[str, Any]):
        """记录日志"""
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "event_type": event_type,
            "data": data
        }
        self.logs.append(log_entry)
        
        # 实时写入文件
        with open(self.log_file, "a", encoding="utf-8") as f:
            f.write(json.dumps(log_entry, ensure_ascii=False) + "\n")
    
    def on_chain_start(self, serialized: Dict[str, Any], 
                      inputs: Dict[str, Any], **kwargs):
        """Chain 开始"""
        chain_id = id(serialized)
        self.start_times[chain_id] = time.time()
        
        self._log("chain_start", {
            "chain_type": serialized.get("name", "unknown"),
            "inputs": {k: str(v)[:200] for k, v in inputs.items()}
        })
    
    def on_chain_end(self, outputs: Dict[str, Any], **kwargs):
        """Chain 结束"""
        self._log("chain_end", {
            "outputs": {k: str(v)[:200] for k, v in outputs.items()}
        })
    
    def on_llm_start(self, serialized: Dict[str, Any], 
                    prompts: List[str], **kwargs):
        """LLM 开始"""
        llm_id = id(serialized)
        self.start_times[llm_id] = time.time()
        
        self._log("llm_start", {
            "llm_type": serialized.get("name", "unknown"),
            "prompt_count": len(prompts),
            "prompts": [p[:200] for p in prompts]
        })
    
    def on_llm_end(self, response: LLMResult, **kwargs):
        """LLM 结束"""
        # 统计 token
        token_usage = {}
        if response.llm_output and "token_usage" in response.llm_output:
            token_usage = response.llm_output["token_usage"]
        
        generations = response.generations
        output_text = generations[0][0].text if generations else ""
        
        self._log("llm_end", {
            "token_usage": token_usage,
            "output_length": len(output_text),
            "output_preview": output_text[:200]
        })
    
    def on_retriever_start(self, serialized: Dict[str, Any], 
                          query: str, **kwargs):
        """检索器开始"""
        self._log("retriever_start", {
            "retriever_type": serialized.get("name", "unknown"),
            "query": query[:200]
        })
    
    def on_retriever_end(self, documents: List[Document], **kwargs):
        """检索器结束"""
        self._log("retriever_end", {
            "document_count": len(documents),
            "documents": [
                {
                    "content": doc.page_content[:100],
                    "metadata": doc.metadata
                }
                for doc in documents[:5]
            ]
        })


class PerformanceCallbackHandler(BaseCallbackHandler):
    """性能监控回调处理器"""
    
    def __init__(self):
        super().__init__()
        self.metrics = {
            "chain_count": 0,
            "llm_count": 0,
            "retriever_count": 0,
            "total_latency": 0,
            "llm_latency": 0
        }
        self.start_times = {}
    
    def on_chain_start(self, serialized: Dict[str, Any], 
                      inputs: Dict[str, Any], **kwargs):
        self.start_times["chain"] = time.time()
        self.metrics["chain_count"] += 1
    
    def on_chain_end(self, outputs: Dict[str, Any], **kwargs):
        if "chain" in self.start_times:
            latency = time.time() - self.start_times["chain"]
            self.metrics["total_latency"] += latency
            del self.start_times["chain"]
    
    def on_llm_start(self, serialized: Dict[str, Any], 
                    prompts: List[str], **kwargs):
        self.start_times["llm"] = time.time()
        self.metrics["llm_count"] += 1
    
    def on_llm_end(self, response: LLMResult, **kwargs):
        if "llm" in self.start_times:
            latency = time.time() - self.start_times["llm"]
            self.metrics["llm_latency"] += latency
            del self.start_times["llm"]
    
    def get_metrics(self) -> Dict[str, Any]:
        """获取性能指标"""
        metrics = self.metrics.copy()
        
        # 计算平均值
        if metrics["chain_count"] > 0:
            metrics["avg_total_latency"] = (
                metrics["total_latency"] / metrics["chain_count"]
            )
        if metrics["llm_count"] > 0:
            metrics["avg_llm_latency"] = (
                metrics["llm_latency"] / metrics["llm_count"]
            )
        
        return metrics
    
    def reset_metrics(self):
        """重置指标"""
        for key in self.metrics:
            self.metrics[key] = 0
        self.start_times.clear()


class TokenUsageCallbackHandler(BaseCallbackHandler):
    """Token 使用统计回调处理器"""
    
    # Token 价格 (每 1K tokens)
    PRICING = {
        "gpt-3.5-turbo": {"input": 0.0015, "output": 0.002},
        "gpt-4": {"input": 0.03, "output": 0.06},
        "deepseek-chat": {"input": 0.001, "output": 0.002},
        "default": {"input": 0.001, "output": 0.002}
    }
    
    def __init__(self, model_name: str = "default"):
        super().__init__()
        self.model_name = model_name
        self.total_prompt_tokens = 0
        self.total_completion_tokens = 0
        self.total_tokens = 0
        self.session_count = 0
    
    def on_llm_end(self, response: LLMResult, **kwargs):
        """记录 Token 使用"""
        if response.llm_output and "token_usage" in response.llm_output:
            usage = response.llm_output["token_usage"]
            
            self.total_prompt_tokens += usage.get("prompt_tokens", 0)
            self.total_completion_tokens += usage.get("completion_tokens", 0)
            self.total_tokens += usage.get("total_tokens", 0)
            self.session_count += 1
    
    def get_usage(self) -> Dict[str, Any]:
        """获取使用统计"""
        pricing = self.PRICING.get(self.model_name, self.PRICING["default"])
        
        input_cost = (self.total_prompt_tokens / 1000) * pricing["input"]
        output_cost = (self.total_completion_tokens / 1000) * pricing["output"]
        
        return {
            "model": self.model_name,
            "total_prompt_tokens": self.total_prompt_tokens,
            "total_completion_tokens": self.total_completion_tokens,
            "total_tokens": self.total_tokens,
            "session_count": self.session_count,
            "estimated_cost_usd": round(input_cost + output_cost, 4),
            "avg_tokens_per_session": (
                self.total_tokens // max(self.session_count, 1)
            )
        }
    
    def reset(self):
        """重置统计"""
        self.total_prompt_tokens = 0
        self.total_completion_tokens = 0
        self.total_tokens = 0
        self.session_count = 0


class CallbackManager:
    """回调管理器"""
    
    def __init__(self):
        self.logging_callback = LoggingCallbackHandler()
        self.performance_callback = PerformanceCallbackHandler()
        self.token_callback = TokenUsageCallbackHandler()
        
        self.all_callbacks = [
            self.logging_callback,
            self.performance_callback,
            self.token_callback
        ]
    
    def get_callbacks(self) -> List[BaseCallbackHandler]:
        """获取所有回调处理器"""
        return self.all_callbacks
    
    def get_metrics(self) -> Dict[str, Any]:
        """获取所有指标"""
        return {
            "performance": self.performance_callback.get_metrics(),
            "token_usage": self.token_callback.get_usage()
        }
    
    def reset(self):
        """重置所有统计"""
        self.performance_callback.reset_metrics()
        self.token_callback.reset()


# 全局回调管理器实例
_callback_manager = None


def get_callback_manager() -> CallbackManager:
    """获取回调管理器 (单例)"""
    global _callback_manager
    if _callback_manager is None:
        _callback_manager = CallbackManager()
    return _callback_manager
```
